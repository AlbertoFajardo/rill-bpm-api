 21 /**
 22  * Repository query
 23  */
 24 var SavedQuery = Backbone.Model.extend({
 25     parse: function(response, XHR) {
 26         this.xml = response.xml;
 27     },
 28     
 29     url: function() {
 30         var segment = Settings.BIPLUGIN ? 
 31                 "/pentahorepository/" : "/repository/";
 32         return encodeURI(Saiku.session.username + segment + this.get('name'));
 33     },
 34     
 35     move_query_to_workspace: function(model, response) {
 36         var query = new Query({ 
 37             xml: model.xml
 38         }, {
 39             name: model.get('name')
 40         });
 41         
 42         var tab = Saiku.tabs.add(new Workspace({ query: query }));
 43     }
 44 });
 45 
 46 /**
 47  * Repository adapter
 48  */
 49 var Repository = Backbone.Collection.extend({
 50     model: SavedQuery,
 51     
 52     initialize: function(args, options) {
 53         this.dialog = options.dialog;
 54     },
 55     
 56     parse: function(response) {
 57         this.dialog.populate(response);
 58     },
 59     
 60     url: function() {
 61         var segment = Settings.BIPLUGIN ? 
 62             "/pentahorepository" : "/repository/";
 63         return encodeURI(Saiku.session.username + segment);
 64     }
 65 });
