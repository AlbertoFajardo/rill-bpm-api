 32 var puc = {
 33     allowSave: function(isAllowed) {
 34         if(window.parent.mantle_initialized !== undefined && window.parent.mantle_initialized && 
 35             window.parent.enableAdhocSave ) {
 36             if (window.ALLOW_PUC_SAVE === undefined || ALLOW_PUC_SAVE) {
 37                 window.parent.enableAdhocSave(isAllowed);
 38             }
 39         }
 40     },
 41 
 42     refresh_repo: function() {
 43         if(window.parent.mantle_initialized !== undefined && window.parent.mantle_initialized) {
 44             window.parent.mantle_refreshRepository();
 45         }
 46     },
 47 
 48     save_to_solution: function(filename, solution, path, type, overwrite) {
 49         var query = Saiku.tabs._tabs[0].content.query;
 50         query.action.get("/xml", {
 51             success: function(model, response) {
 52                 (new SavedQuery({
 53                     name: filename,
 54                     newname: query.get('name'),
 55                     xml: response.xml,
 56                     solution: solution,
 57                     path: path,
 58                     action: filename,
 59                     overwrite: overwrite
 60                 })).save({ success: function() {
 61                     puc.refresh_repo();
 62                 }});
 63             }
 64         });
 65     }
 66 };

100 /**
101  * Bind callbacks to workspace
102  */
103 var BIPlugin = {
104     bind_callbacks: function(workspace) {
105         // If in view mode, remove sidebar and drop zones
106         if (Settings.MODE == "view" || Settings.MODE == "table") {
107             workspace.toggle_sidebar();
108             $(workspace.el).find('.sidebar_separator').remove();
109             $(workspace.el).find('.workspace_inner')
110                 .css({ 'margin-left': 0 });
111             $(workspace.el).find('.workspace_fields').remove();
112         }
113 
114         // Remove toolbar buttons
115         $(workspace.toolbar.el)
116             .find('.save').parent().remove();
117         $(workspace.toolbar.el).find('.run').parent().removeClass('seperator');
118         if (Settings.MODE == "view" || Settings.MODE == "table") {
119             $(workspace.toolbar.el)
120                 .find(".run, .auto, .toggle_fields, .toggle_sidebar")
121                 .parent().remove();
122         }
123         if (Settings.MODE == "table") {
124             $(workspace.toolbar.el).parent().remove();
125         }
126 
127         // Toggle save button
128         workspace.bind('query:result', function(args) {
129             var isAllowed = args.data.cellset && 
130                 args.data.cellset.length > 0;
131             puc.allowSave(isAllowed);
132         });
133     }
134 };

139 Saiku.events.bind('session:new', function(session) {

145         if (Saiku.tabs._tabs[0] && Saiku.tabs._tabs[0].content) {
146             BIPlugin.bind_callbacks(Saiku.tabs._tabs[0].content);
147         }
148 
149         Saiku.session.bind('workspace:new', function(args) {
150             BIPlugin.bind_callbacks(args.workspace);
151         });
152     }

177     window.parent.getSaikuMdx = function() {
178             var myself = this;
179             var query = Saiku.tabs._tabs[0].content.query;
180             query.clear();
181             query.fetch({ 
182             success: function(model, response) {
183                     var ds = new Datasources();
184                     ds.fetch({
185                         success: function(dmodel, dresponse) {
186                             for (var i = 0; i < dresponse.length; i ++) {
187                                 if (dresponse[i].name == response.cube.connectionName) {
188                                     var urlParts = dresponse[i].properties.location.split(';');
189                                     var jndi = "";
190                                     var catalog = "";
191                                     $.each(urlParts,function(index, value) { 
192                                         var prop = value.split('=');
193                                         if (prop[0] == "DataSource") {
194                                             jndi = prop[1];
195                                         }
196                                        if (prop[0] == "Catalog") {
197                                             catalog = prop[1];
198                                         } 
199                                     });
200                                     var saikuStub = {
201                                         connection: dresponse[i].name,
202                                         catalog: catalog,
203                                         jndi: jndi,
204                                         mdx: response.mdx
205                                     }
206                                     window.parent.saveSaiku(saikuStub);
207                                 }        
208                             }
209                         }
210                     })
211                     
212                 }
213             });
214     };
