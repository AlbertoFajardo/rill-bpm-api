 21 /**
 22  * Model which fetches the dimensions for a cube
 23  */
 24 var Dimension = Backbone.Model.extend({
 25     initialize: function(args) {
 26         this.url = Saiku.session.username + "/discover/" +
 27             args.key + "/dimensions";
 28         this.clear();
 29     },
 30     
 31     parse: function(response) {
 32         this.set({
 33             template: _.template($("#template-dimensions").html())({
 34                 dimensions: response
 35             }),
 36             
 37             data: response
 38         });
 39         
 40         localStorage && localStorage.setItem("dimension." + this.get('key'),
 41                 JSON.stringify(this));
 42         
 43         return response;
 44     }
 45 });
