 22 /**
 23  * Renders a stats for each workspace
 24  */
 25 var Statistics = Backbone.View.extend({
 26     initialize: function(args) {
 27         this.workspace = args.workspace;
 28         
 29         // Create a unique ID for use as the CSS selector
 30         this.id = _.uniqueId("stats_");
 31         $(this.el).attr({ id: this.id });
 32         
 33         // Bind table rendering to query result event
 34         _.bindAll(this, "render", "receive_data", "process_data", "show", 
 35             "setOptions");
 36         this.workspace.bind('query:result', this.receive_data);
 37         
 38         // Add stats button
 39         this.add_button();
 40         this.workspace.toolbar.stats = this.show;
 41         
 42         // Listen to adjust event and rerender stats
 43         this.workspace.bind('workspace:adjust', this.render);
 44         
 45         // Append stats to workspace
 46         $(this.workspace.el).find('.workspace_results')
 47             .prepend($(this.el).hide())
 48     },
 49     
 50     add_button: function() {
 51         var $stats_button = 
 52             $('<a href="#stats" class="stats button disabled_toolbar i18n" title="Basic Statistics"></a>')
 53             .css({  'background-image': "url('js/saiku/plugins/Statistics/sigma.png')",
 54                     'background-repeat':'no-repeat',
 55                     'background-position':'50% 50%'
 56                 });
 57 
 58         var $stats_li = $('<li class="seperator"></li>').append($stats_button);
 59         $(this.workspace.toolbar.el).find("ul").append($stats_li);
 60     },
 61     
 62     show: function(event, ui) {
 63         $(this.workspace.el).find('.workspace_results table').toggle();
 64         $(this.el).toggle();
 65         $(event.target).toggleClass('on');
 66         
 67         if ($(event.target).hasClass('on')) {
 68             this.render();
 69         }
 70     },
 71     
 72     setOptions: function(event) {
 73         var type = $(event.target).attr('href').replace('#', '');
 74         try {
 75             this[type]();
 76         } catch (e) { }
 77         
 78         return false;
 79     },
 80     
 81     render: function() {
 82         if (! $(this.workspace.toolbar.el).find('.stats').hasClass('on')) {
 83             return;
 84         }
 85 
 86         var createRow = function(cells, header){
 87             var row = "<tr>"
 88             var ohcell = header?"<th class='row_header'>":"<th class='row'>"
 89             var odcell = header?"<th class='col'>":"<td class='data'>"
 90             var ccell = header?"</th>":"</td>"
 91             _.each(cells, function(it, idx){row += ((idx==0)?ohcell:odcell) + "<div>" + it + "</div>" + ccell})
 92             row += "</tr>"
 93             return row
 94         }
 95 
 96         var group = function(grid, el, cback){
 97             var elements = _.filter(_.map(grid, function(it){return it[el]}), function(it){return it})
 98             return cback(elements).toFixed(3)
 99         }
100 
101         var sum = function(elems){return _.reduce(elems, function(memo, num){ return memo + num }, 0)}
102         var average = function(elems){return (sum(elems))/elems.length}
103         var stdx = function(elems){
104             var m = average(elems)
105             return Math.sqrt(sum(_.map(elems, function(it){return Math.pow(m - it,2)}))/elems.length)
106         }
107         var min = function(elems){return _.min(elems)}
108         var max = function(elems){return _.max(elems)}
109         
110         $(this.el).empty()
111         var grid = this.data.metadata
112         var rs = this.data.resultset
113         var aux = _.filter(grid, function(it){return !it.isHeader && it.colType == 'Numeric'})
114         var columns = _.map(aux, function(it){return it.colName})
115         var idxs =_.map(columns, function(el){return _.indexOf(_.map(grid, function(it){return it.colName}), el)})
116         
117         var $table = $("<table style='display: table; '>")
118         var $tbody = $("<tbody>").appendTo($table)
119         $tbody.append(createRow(['Statistics'].concat(columns), true))
120         $tbody.append(createRow(['Min'].concat(_.map(idxs, function(it){return group(rs, it, min)})), false))
121         $tbody.append(createRow(['Max'].concat(_.map(idxs, function(it){return group(rs, it, max)})), false))
122         $tbody.append(createRow(['Sum'].concat(_.map(idxs, function(it){return group(rs, it, sum)})), false))
123         $tbody.append(createRow(['Average'].concat(_.map(idxs, function(it){return group(rs, it, average)})), false))
124         $tbody.append(createRow(['Std. Deviation'].concat(_.map(idxs, function(it){return group(rs, it, stdx)})), false))
125 
126         $(this.el).append($table)
127         
128     },
129     
130     receive_data: function(args) {
131         return _.delay(this.process_data, 0, args);
132     },
133     
134     process_data: function(args) {
135         this.data = {};
136         this.data.resultset = [];
137         this.data.metadata = [];
138         this.data.height = 0;
139         this.data.width = 0;
140 
141         if (args.data.cellset && args.data.cellset.length > 0) {
142             
143             var lowest_level = 0;
144             var isHead = true
145             var columnNames = new Array()
146             for (var row = 0; row < args.data.cellset.length; row++) {
147                 if (isHead && (args.data.cellset[row][0].type == "ROW_HEADER_HEADER" || 
148                     args.data.cellset[row][0].value == "null")) {
149                     this.data.metadata = [];
150                     for (var field = 0; field < args.data.cellset[row].length; field++) {
151                         if (args.data.cellset[row][field].type == "ROW_HEADER_HEADER") {
152                             this.data.metadata.shift();
153                             lowest_level = field;
154                         }
155                         if(columnNames[field]){
156                             columnNames[field].push(args.data.cellset[row][field].value)
157                         }else{
158                             columnNames[field] = [args.data.cellset[row][field].value]
159                         }
160                         if(args.data.cellset[row][0].type == "ROW_HEADER_HEADER"){
161                             this.data.metadata.push({
162                                 colIndex: field,
163                                 colType: typeof(args.data.cellset[row + 1][field].value) !== "number" &&
164                                     isNaN(args.data.cellset[row + 1][field].value
165                                     .replace(/[^a-zA-Z 0-9.]+/g,'')) ? "String" : "Numeric",
166                                 colName: columnNames[field].join(' / '),
167                                 isHeader: (args.data.cellset[row][field].type == "ROW_HEADER_HEADER")
168                             });    
169                         }
170                     }
171                 } else if (args.data.cellset[row][lowest_level].value !== "null" && args.data.cellset[row][lowest_level].value !== "") {
172                     isHead = false
173                     var record = [];
174                     this.data.width = args.data.cellset[row].length;
175                     for (var col = lowest_level; col < args.data.cellset[row].length; col++) {
176                         var value = args.data.cellset[row][col].value;
177                         // check if the resultset contains the raw value, if not try to parse the given value
178                         if (args.data.cellset[row][col].properties.raw && args.data.cellset[row][col].properties.raw !== "null")
179                         {
180                             value = parseFloat(args.data.cellset[row][col].properties.raw);
181                         } else if (typeof(args.data.cellset[row][col].value) !== "number" &&
182                             parseFloat(args.data.cellset[row][col].value.replace(/[^a-zA-Z 0-9.]+/g,''))) 
183                         {
184                             value = parseFloat(args.data.cellset[row][col].value.replace(/[^a-zA-Z 0-9.]+/g,''));
185                         }
186                         if (col == lowest_level) {
187                             value += " [" + row + "]";
188                         }
189                         record.push(value);
190                     }
191                     this.data.resultset.push(record);
192                 }
193             }
194             this.data.height = this.data.resultset.length;
195             this.render();
196         } else {
197             $(this.el).text("No results");
198         }
199     }
200 });

202 /**
203  * Start Plugin
204  */ 
205  Saiku.events.bind('session:new', function(session) {
206 
207         function new_workspace(args) {
208             // Add stats element
209             if (typeof args.workspace.stats == "undefined") {
210                 args.workspace.stats = new Statistics({ workspace: args.workspace });
211             }
212         }
213 
214         function clear_workspace(args) {
215             if (typeof args.workspace.stats != "undefined") {
216                 $(args.workspace.stats.el).parents().find('.workspace_results table').show();
217                 $(args.workspace.stats.el).hide();
218             }
219         }
220 
221         
222         // Attach stats to existing tabs
223         for(var i = 0; i < Saiku.tabs._tabs.length; i++) {
224             var tab = Saiku.tabs._tabs[i];
225             new_workspace({
226                 workspace: tab.content
227             });
228         };
229 
230         // Attach stats to future tabs
231         Saiku.session.bind("workspace:new", new_workspace);
232         Saiku.session.bind("workspace:clear", clear_workspace);
233     });
