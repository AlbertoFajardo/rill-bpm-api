 21 module('Tab', {
 22     setup: function() {
 23         var workspace = new Workspace();
 24         this.tab = new Tab({ content: workspace });
 25     }
 26 });
 27 
 28 test('event handler', function event_handler() {
 29     var global_value = false;
 30     this.tab.bind('zero', function() {
 31         global_value = true;
 32     });
 33     
 34     this.tab.trigger('zero');
 35     equal(true, global_value, 'Could not bind event listener');
 36 });
 37 
 38 test('multiple event handlers', function multiple_handlers() {
 39     var global_value = 0;
 40     
 41     this.tab.bind('multiple', function() {
 42         global_value+=1;
 43     });
 44     
 45     this.tab.bind('multiple', function() {
 46         global_value+=1;
 47     });
 48     
 49     this.tab.bind('multiple', function() {
 50         global_value+=1;
 51     });
 52     
 53     this.tab.trigger('multiple');
 54     equal(3, global_value, 'Could not bind multiple event listeners');
 55 });
 56 
 57 test('tab render', function tab_render() {
 58     this.tab.render();
 59     notStrictEqual(-1, $(this.tab.el).html().indexOf('Unsaved query'));
 60     strictEqual(-1, $(this.tab.el).html().indexOf('Unsaved Query (1)'));
 61 });
 62 
 63 test('tab properties', function tab_properties() {
 64     notEqual(this.tab.caption, "");
 65     notStrictEqual(this.tab.id, undefined);
 66     equal(this.tab.id, this.tab.content.tab.id);
 67     notStrictEqual(undefined, this.tab.bind);
 68     notStrictEqual(undefined, this.tab.trigger);
 69 });

 71 module('TabSet', {
 72     setup: function() {
 73         this.tabset = new TabSet();
 74     },
 75     
 76     teardown: function() {
 77         $("#tab_panel").remove();
 78     }
 79 });
 80 
 81 test('initial conditions', function initial_conditions() {
 82     equal(this.tabset.queryCount, 0);
 83     equal(undefined, this.tabset.pager);
 84     this.tabset.render();
 85     ok(this.tabset.pager !== undefined);
 86     ok(this.tabset.content !== undefined);
 87     notEqual(-1, $(this.tabset.el).html().indexOf('#pager'));
 88 });
 89 
 90 test('tab lifecycle', function add_tab() {
 91     // Add tab
 92     this.tabset.render();
 93     raises(this.tabset.add, TypeError, 
 94         "Don't allow tab instantiation without Backbone view as parameter");
 95     var tab = this.tabset.add(new Workspace());
 96     ok(this.tabset === tab.parent, 
 97         "Should set parent property when creating tab");
 98     equal(1, this.tabset.queryCount,
 99         "Query count should be incremented");
100     equal(this.tabset.queryCount, this.tabset._tabs.length,
101         "Because no tabs have been removed, query count and tab length should match");
102     var tab2 = this.tabset.add(new Workspace());
103     equal(2, this.tabset.queryCount,
104         "Query count should be incremented");
105     
106     // Select tab
107     this.tabset.select(tab);
108     this.tabset.select(tab2);
109     
110     // Remove tab
111     this.tabset.remove(tab);
112     equal(2, this.tabset.queryCount,
113         "Query count should not decrement");
114     notEqual(this.tabset.queryCount, this.tabset._tabs.length,
115         "Query count should not equal tab length");
116     this.tabset.remove(tab2);
117     equal(1, this.tabset._tabs.length,
118         "Tabset should add a tab when empty");
119 });
