 21 /**
 22  * The open query tab (Repository viewer)
 23  */
 24 var OpenQuery = Backbone.View.extend({
 25     className: 'tab_container',
 26     
 27     events: {
 28         'click #queries li': 'view_query',
 29         'dblclick #queries li': 'select_and_open_query',
 30         'click .workspace_toolbar a.open': 'open_query',
 31         'click .workspace_toolbar a.delete': 'delete_query'
 32     },
 33     
 34     template: function() {
 35         return _.template($("#template-open-dialog").html())();        
 36     },
 37     
 38     caption: function() {
 39         return "Repository";
 40     },
 41     
 42     render: function() {
 43         // Load template
 44         $(this.el).html(this.template());
 45         
 46         // Adjust tab when selected
 47         this.tab.bind('tab:select', this.fetch_queries);
 48         this.tab.bind('tab:select', this.adjust);
 49         $(window).resize(this.adjust);
 50         
 51         return this; 
 52     },
 53     
 54     initialize: function(args) {
 55         // Maintain `this`
 56         _.bindAll(this, "adjust", "fetch_queries",
 57                 "clear_query","select_and_open_query");
 58         
 59         // Initialize repository
 60         this.repository = new Repository({}, { dialog: this });
 61     },
 62     
 63     fetch_queries: function() {
 64         this.repository.fetch();
 65     },
 66     
 67     populate: function(response) {
 68         this.queries = {};
 69         var $ul = $(this.el).find('.sidebar ul').html('');
 70         for (var i = 0; i < response.length; i++) {
 71             var query = response[i];
 72             this.queries[query.name] = query;
 73             var $link = $("<a />").text(query.name)
 74                 .attr({ href: "#" + query.name });
 75             var $icon = $("<span class='sprite'></span>");
 76             $("<li />").append($icon)
 77                 .append($link)
 78                 .appendTo($ul);
 79         }
 80     },
 81     
 82     view_query: function(event) {
 83         $target = $(event.currentTarget).find('a');
 84         var name = $target.attr('href').replace('#', '');
 85         var query = this.queries[name];
 86         
 87         $(this.el).find('.workspace_toolbar').removeClass('hide');
 88         
 89         $results = $(this.el).find('.workspace_results')
 90             .html('<h3><strong>' + query.name + '</strong></h3>');
 91         var $properties = $('<ul id="query_info" />').appendTo($results);
 92         
 93         // Iterate through properties and show a key=>value set in the information pane
 94         for (var property in query) {
 95             if (query.hasOwnProperty(property) && property != "name") {
 96                 $properties.append($('<li />').html("<strong>" + 
 97                         property + "</strong> : " + query[property]));
 98             }
 99         }
100         
101         this.selected_query = new SavedQuery({ name: name });
102         
103         return false;
104     },
105 
106     select_and_open_query: function(event) {
107         $target = $(event.currentTarget).find('a');
108         var name = $target.attr('href').replace('#', '');
109         this.selected_query = new SavedQuery({ name: name });
110         this.open_query();
111     },
112     
113     open_query: function(event) {
114         Saiku.ui.block("Opening query...");
115         this.selected_query.fetch({ 
116             success: this.selected_query.move_query_to_workspace 
117         });
118         
119         return false;
120     },
121     
122     delete_query: function(event) {
123         (new DeleteQuery({
124             query: this.selected_query,
125             success: this.clear_query
126         })).render().open();
127         
128         return false;
129     },
130     
131     clear_query: function() {
132         $(this.el).find('.workspace_toolbar').addClass('hide');
133         $(this.el).find('.workspace_results').html('');
134         this.fetch_queries();
135     },
136     
137     adjust: function() {
138         // Adjust the height of the separator
139         $separator = $(this.el).find('.sidebar_separator');
140         $separator.height($("body").height() - 87);
141         $(this.el).find('.sidebar').height($("body").height() - 87);
142         
143         // Adjust the dimensions of the results window
144         $(this.el).find('.workspace_results').css({
145             width: $(document).width() - $(this.el).find('.sidebar').width() - 30,
146             height: $(document).height() - $("#header").height() -
147                 $(this.el).find('.workspace_toolbar').height() - 
148                 $(this.el).find('.workspace_fields').height() - 40
149         });
150     },
151     
152     toggle_sidebar: function() {
153         // Toggle sidebar
154         $(this.el).find('.sidebar').toggleClass('hide');
155         var new_margin = $(this.el).find('.sidebar').hasClass('hide') ?
156                 5 : 265;
157         $(this.el).find('.workspace_inner').css({ 'margin-left': new_margin });
158     }
159 });
