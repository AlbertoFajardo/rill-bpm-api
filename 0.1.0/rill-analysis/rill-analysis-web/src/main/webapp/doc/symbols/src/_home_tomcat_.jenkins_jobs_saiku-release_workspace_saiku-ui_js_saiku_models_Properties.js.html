 21 /**
 22  * Maintains query properties like non empty and automatic execution
 23  */
 24 var Properties = Backbone.Model.extend({    
 25     initialize: function(args, options) {
 26         // Keep track of parent query
 27         this.query = options.query;
 28         
 29         // Update properties with defaults from settings
 30         this.properties = {};
 31         _.extend(this.properties, Settings.QUERY_PROPERTIES);
 32         this.update();
 33     },
 34     
 35     toggle: function(key) {
 36         // Toggle property
 37         this.properties[key] = this.properties[key] === 'true' ? 
 38             'false' : 'true';
 39         
 40         return this;
 41     },
 42     
 43     update: function() {
 44         // FIXME - this really sucks
 45         // Why can't we just use the body?
 46         this.attributes = {
 47             properties: _.template(
 48                     "<% _.each(properties, function(property, name) { %>" +
 49                     "<%= name %> <%= property %>\n" +
 50                     "<% }); %>"
 51                     )({ properties: this.properties })
 52         };
 53         this.save();
 54     },
 55     
 56     parse: function(response) {
 57         // FIXME - POST should return properties as well
 58         if (typeof response == "object") {
 59             _.extend(this.properties, response);
 60         }
 61         
 62         this.query.workspace.trigger('properties:loaded');
 63     },
 64     
 65     url: function() {
 66         return encodeURI(this.query.url() + "/properties");
 67     }
 68 });
