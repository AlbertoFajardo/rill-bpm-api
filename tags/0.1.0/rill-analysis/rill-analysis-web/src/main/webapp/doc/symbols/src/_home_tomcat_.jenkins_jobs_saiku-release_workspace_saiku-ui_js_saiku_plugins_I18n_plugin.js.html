 21 /**
 22  * The user's current locale
 23  */
 24 Saiku.i18n = {
 25     locale: (navigator.language || navigator.browserLanguage || 
 26         navigator.systemLanguage || navigator.userLanguage).substring(0,2).toLowerCase(),
 27     po_file: {},
 28     translate: function() {
 29         $('.i18n').i18n(Saiku.i18n.po_file);
 30     },
 31     automatic_i18n: function() {
 32         // Load language file if it isn't English
 33         if (Saiku.i18n.locale != "en") {
 34             $.ajax({
 35                 url: "js/saiku/plugins/I18n/po/" + Saiku.i18n.locale + ".json",
 36                 type: 'GET',
 37                 dataType: 'json',
 38                 success: function(data) {
 39                     Saiku.i18n.po_file = data;
 40                     Saiku.i18n.translate();
 41                 }
 42             });
 43         }
 44         
 45         return true;
 46     },
 47     elements: [],
 48     improve_translation: function() {
 49         Saiku.tabs.add(new TranslationTab());
 50         return false;
 51     }
 52 };

 54 /**
 55  * jQuery plugin for i18n
 56  */
 57 (function( $ ){
 58 	/**
 59 	 * Internationalize selected elements with the provided PO file
 60 	 */
 61 	$.fn.i18n = function(po_file) {
 62 		// If no PO file is provided, then don't translate anything
 63 		if (! po_file)
 64 			return this;
 65 		
 66 		// If key is not found, return original language
 67 		var translate = function(key, po_file) {
 68 			if (typeof po_file[key] == "undefined") {
 69 				return "";
 70 			} else {
 71 				return po_file[key];
 72 			}
 73 		};		
 74 
 75 		// Iterate over UI elements that need to be translated
 76 		return $.each(this, function() {
 77 			element = $(this);
 78 			
 79 			// Translate text
 80 			if (element.html()) {
 81 				translated_text = translate( element.html(), po_file );
 82                 if (Saiku.i18n.elements.indexOf &&
 83                     Saiku.i18n.elements.indexOf(element.html()) === -1) {
 84                     Saiku.i18n.elements.push(element.html());
 85                 }
 86 				if (translated_text) {
 87 					element.data('original', element.html());
 88 					element.html(translated_text);
 89 					element.removeClass('i18n');
 90 				}
 91 			}
 92 			
 93 			// Translate title
 94 			if (element.attr('title')) {
 95 				translated_title = translate( element.attr('title'), po_file );
 96                 if (Saiku.i18n.elements.indexOf && 
 97                     Saiku.i18n.elements.indexOf(element.attr('title')) === -1) {
 98                     Saiku.i18n.elements.push(element.attr('title'));
 99                 }
100 				if (translated_title) {
101 					element.data('original', element.attr('title'));
102 					element.attr({ 'title': translated_title });
103 					element.removeClass('i18n');
104 				}
105 			}
106 			
107 			// Remove class so this element isn't repeatedly translated
108 			if (element.hasClass('i18n')) {
109 			    element.addClass('i18n_failed');
110 			}
111 			element.addClass('i18n_translated');
112 		});
113 	};
114 	
115 	$.fn.un_i18n = function() {
116 		// Iterate over UI elements to replace the original text
117 		return $.each(this, function() {
118 			element = $(this);
119 			
120 			if (element.text())
121 				element.text(element.data('original'));
122 			
123 			if (element.attr('title'))
124 				element.attr({ 'title': element.data('original') });
125 			
126 			element.addClass('i18n');
127 			element.removeClass('i18n_translated')
128 			    .removeClass('i18n_failed');
129 		});
130 	};
131 })( jQuery );

133 /**
134  * Backbone view which allows users to submit translations
135  */
136 var TranslationTab = Backbone.View.extend({
137     className: 'workspace_area',
138     caption: function() {
139         return Saiku.i18n.po_file["Improve this translation"] ?
140             Saiku.i18n.po_file["Improve this translation"] :
141             "Improve this translation";
142     },
143     events: {
144         'submit form': 'submit',
145         'change input': 'mark'
146     },
147     initialize: function() {
148         $(window).resize(this.adjust);
149         $(this.el).focus(this.adjust);
150         this.adjust();
151     },
152     render: function() {
153         var translation_table = {};
154         for (var i = 0; i < Saiku.i18n.elements.length; i++) {
155             translation_table[Saiku.i18n.elements[i]] = {
156                 value: Saiku.i18n.po_file[Saiku.i18n.elements[i]],
157                 name: encodeURI(Saiku.i18n.elements[i])
158             };
159         }
160         var table = _.template("<form class='workspace_results'>" +
161         	"Your name: <input type='text' name='translator_name' />" +
162             "<p>Please fill in the appropriate translation in the blanks provided:<p>" +
163         	"<% _.each(translation_table, function(val, key) { %>" +
164             "<div><b><%= key %></b><br />" +
165             "<input type='text' value='<%= val.value %>' name='<%= val.name %>' />" +
166             "</div>" +
167             "<% }); %>" +
168             "<div><input class='submit-translation' type='submit' value='Submit translation' /></div>" +
169             "</form>")({
170                 translation_table: translation_table
171             });
172         $(this.el).html(table).find('div').css({
173             'float': 'left',
174             'padding': '20px'
175         }).find('input').css({
176             'width': '300px'
177         });
178         $(this.el).find('.submit-translation').css({
179             'padding': '20px'
180         });
181     },
182     
183     mark: function(event) {
184         $(event.target).addClass('changed');
185     },
186     
187     submit: function() {
188         var translation = { locale: Saiku.i18n.locale };
189         $(this.el).find('.changed').each(function(element) {
190             translation[decodeURI($(this).attr('name'))] = encodeURI($(this).val());
191         });
192         Translate.log(translation);
193         Saiku.ui.block('Thank you for improving our translation!');
194         this.tab.remove();
195         _.delay(function() {
196             Saiku.ui.unblock();
197         }, 1000);
198         return false;
199     },
200     
201     adjust: function() {
202         $(this.el).height($("body").height() - 87);
203     }
204 });

211 /** 
212  * Add translate button
213  */
214 Saiku.events.bind('toolbar:render', function(args) {
215     if (Saiku.i18n.locale != "en") {
216         var $link = $("<a />").text(Saiku.i18n.locale)
217             .attr({ 
218                 href: "#translate",
219                 title: "Improve this translation"
220             })
221             .click(Saiku.i18n.improve_translation)
222             .addClass('sprite translate i18n');
223         var $li = $("<li />").append($link);
224         $(args.toolbar.el).find('ul').append($li);
225     }
226 });

228 /**
229  * Bind to new workspace
230  */
231 Saiku.events.bind('session:new', function() {    
232     // Translate elements already rendered
233     Saiku.i18n.translate();
234     
235     // Translate new workspaces
236     Saiku.session.bind('tab:add', Saiku.i18n.translate);
237 });
