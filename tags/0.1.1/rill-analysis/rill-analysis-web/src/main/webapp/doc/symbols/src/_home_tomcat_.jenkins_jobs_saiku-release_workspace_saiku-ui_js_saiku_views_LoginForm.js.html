 21 /**
 22  * The login prompt on startup
 23  */
 24 var LoginForm = Modal.extend({
 25     type: "login",
 26     message: "<form id='login_form'>" +
 27         "<label for='username'>Username</label><br />" +
 28         "<input type='text' id='username' name='username' value='' /><br />" +
 29         "<label for='password'>Password</label><br />" +
 30         "<input type='password' id='password' name='password' value='' />" +
 31         "</form>",
 32         
 33     buttons: [
 34         { text: "Login", method: "login" }
 35     ],
 36     
 37     events: {
 38         'click a': 'call',
 39         'keyup #login_form input': 'check'
 40     },
 41     
 42     initialize: function(args) {
 43         _.extend(this, args);
 44         _.bindAll(this, "adjust");
 45         this.options.title = Settings.VERSION;
 46         this.bind('open', this.adjust);
 47     },
 48     
 49     adjust: function() {
 50         $(this.el).parent().find('.ui-dialog-titlebar-close').hide();
 51         $(this.el).find("#username").select().focus();
 52     },
 53     
 54     check: function(event) {
 55         if(event.which === 13) {
 56             this.login();
 57         }
 58     },
 59     
 60     login: function() {
 61         
 62         var l_username = $(this.el).find("#username").val();
 63         var l_password = $(this.el).find("#password").val();
 64         $(this.el).dialog('close');
 65         this.session.login(l_username, l_password);
 66 
 67         return true;
 68     }
 69 });
