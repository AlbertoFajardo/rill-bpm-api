 21 /**
 22  * Class which handles table rendering of resultsets
 23  */
 24 var Table = Backbone.View.extend({
 25     tagName: "table",
 26 
 27     events: {
 28         'click .cancel': 'cancel'
 29     },
 30 
 31     initialize: function(args) {
 32         this.workspace = args.workspace;
 33         
 34         // Bind table rendering to query result event
 35         _.bindAll(this, "render", "process_data", "cancelled", "cancel");
 36         this.workspace.bind('query:result', this.render);
 37     },
 38     
 39     render: function(args, block) {
 40 
 41         if (args.data.error != null) {
 42             return this.error(args);
 43         }
 44 
 45         
 46         // Check to see if there is data
 47         if (args.data.cellset && args.data.cellset.length === 0) {
 48             return this.no_results(args);
 49         }
 50         
 51         // Clear the contents of the table
 52         $(this.el).html('<tr><td>Rendering ' + args.data.width + ' columns and ' + args.data.height + ' rows...</td></tr>');
 53 
 54         // Render the table without blocking the UI thread
 55         if (block === true) {
 56             this.process_data(args.data.cellset);
 57         } else {
 58             _.delay(this.process_data, 0, args.data.cellset);
 59         }
 60 
 61     },
 62     
 63     process_data: function(data) {
 64         var contents = "";
 65         var table = data ? data : [];
 66         for (var row = 0; row < table.length; row++) {
 67             contents += "<tr>";
 68             for (var col = 0; col < table[row].length; col++) {
 69                 var header = data[row][col];
 70                 
 71                 // FIXME - this needs to be cleaned up
 72                 
 73                 // If the cell is a column header and is null (top left of table)
 74                 if (header.type === "COLUMN_HEADER" && header.value === "null") {
 75                     contents += '<th class="all_null"><div> </div></th>';
 76                 } // If the cell is a column header and isn't null (column header of table)
 77                 else if (header.type === "COLUMN_HEADER") {
 78                     contents += '<th class="col"><div>' + header.value + '</div></th>';
 79                 } // If the cell is a row header and is null (grouped row header)
 80                 else if (header.type === "ROW_HEADER" && header.value === "null") {
 81                     contents += '<th class="row_null"><div> </div></th>';
 82                 } // If the cell is a row header and isn't null (last row header)
 83                 else if (header.type === "ROW_HEADER") {
 84                     contents += '<th class="row"><div>' + header.value + '</div></th>';
 85                 }
 86                 else if (header.type === "ROW_HEADER_HEADER") {
 87                     contents += '<th class="row_header"><div>' + header.value + '</div></th>';
 88                 } // If the cell is a normal data cell
 89                 else if (header.type === "DATA_CELL") {
 90                     contents += '<td class="data"><div alt="' + header.properties.raw + '" rel="' + header.properties.position + '">' + header.value + '</div></td>';
 91                 }
 92             }
 93             contents += "</tr>";
 94         }
 95         
 96         // Append the table
 97         $(this.el).html(contents);
 98         
 99     },
100     cancel: function(event) {
101         this.workspace.query.action.del("/result", {success: this.cancelled } );
102     },
103     
104     cancelled: function(args) {
105         $(this.el).html('<tr><td>No results</td></tr>');
106     },
107 
108     no_results: function(args) {
109         $(this.el).html('<tr><td>No results</td></tr>');
110     },
111     
112     error: function(args) {
113         $(this.el).html('<tr><td>' + args.data.error + '</td></tr>');
114     }
115 });
