 21 var QueryScenario = Backbone.Model.extend({
 22     initialize: function(args, options) {
 23         // Maintain `this`
 24         _.bindAll(this, "attach_listeners", "activate", "clicked_cell", "save_writeback", 
 25             "cancel_writeback", "check_input");
 26         
 27         this.query = options.query;
 28     },
 29     
 30     activate: function() {
 31         $(this.query.workspace.el).find("td.data").unbind('click').addClass('cellhighlight').click(this.clicked_cell);
 32     },
 33 
 34     attach_listeners: function(args) {
 35         if (args.workspace.query && args.workspace.query.properties &&
 36             args.workspace.query.properties.properties['org.saiku.connection.scenario'] === "true" &&
 37             $(args.workspace.el).find('.query_scenario').hasClass('on'))
 38         $(args.workspace.el).find("td.data").click(this.clicked_cell);
 39     },
 40     
 41     clicked_cell: function(event) {
 42         $target = $(event.target).hasClass('data') ?
 43             $(event.target).find('div') : $(event.target);
 44         var value = $target.attr('alt');
 45         var pos = $target.attr('rel');
 46         
 47         var $input = $("<input type='text' value='" + value + "' />")
 48             .keyup(this.check_input)
 49             .blur(this.cancel_writeback);
 50         $target.html('').append($input);
 51         $input.focus();
 52     },
 53     
 54     check_input: function(event) {
 55         if (event.which == 13) {
 56             this.save_writeback(event);
 57         } else if (event.which == 27 || event.which == 9) {
 58             this.cancel_writeback(event);
 59         }
 60          
 61         return false;
 62     },
 63     
 64     save_writeback: function(event) {
 65         var $input = $(event.target).closest('input');
 66         this.set({
 67             value: $input.val(),
 68             position: $input.parent().attr('rel')
 69         });
 70         this.save();
 71         var value = $input.val();
 72         $input.parent().text(value);
 73     },
 74     
 75     cancel_writeback: function(event) {
 76         var $input = $(event.target).closest('input');
 77         $input.parent().text($input.parent().attr('alt'));
 78     },
 79     
 80     parse: function() {
 81         this.query.run();
 82     },
 83 
 84     url: function() {
 85         return this.query.url() + "/cell/" + this.get('position') + 
 86             "/" + this.get('value'); 
 87     }
 88 });
