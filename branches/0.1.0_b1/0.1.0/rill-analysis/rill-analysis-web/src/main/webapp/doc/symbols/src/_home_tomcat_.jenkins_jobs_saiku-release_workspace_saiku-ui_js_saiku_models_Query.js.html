 21 /**
 22  * Workspace query
 23  */
 24 var Query = Backbone.Model.extend({
 25     initialize: function(args, options) {
 26         // Save cube
 27         _.extend(this, options);
 28         
 29         // Bind `this`
 30         _.bindAll(this, "run", "move_dimension", "reflect_properties");
 31         
 32         // Generate a unique query id
 33         this.uuid = 'xxxxxxxx-xxxx-xxxx-yxxx-xxxxxxxxxxxx'.replace(/[xy]/g, 
 34             function (c) {
 35                 var r = Math.random() * 16 | 0,
 36                 v = c == 'x' ? r : (r & 0x3 | 0x8);
 37                 return v.toString(16);
 38             }).toUpperCase();
 39         
 40         // Initialize properties, action handler, and result handler
 41         this.action = new QueryAction({}, { query: this });
 42         this.result = new Result({}, { query: this });
 43         this.scenario = new QueryScenario({}, { query: this });
 44     },
 45     
 46     parse: function(response) {
 47         // Assign id so Backbone knows to PUT instead of POST
 48         this.id = this.uuid;
 49 
 50         // Grab attributes
 51         if (this.attributes.xml !== undefined && 
 52             this.attributes.cube === undefined) {            
 53             this.set({
 54                 connection: response.cube.connectionName,
 55                 catalog: response.cube.catalogName,
 56                 schema: response.cube.schemaName,
 57                 cube: encodeURIComponent(response.cube.name),
 58                 axes: response.saikuAxes,
 59                 type: response.type
 60             });
 61         }
 62 
 63         // Fetch initial properties from server
 64         if (! this.properties) {
 65             this.properties = new Properties({}, { query: this });
 66         } else {
 67             this.properties.fetch({
 68                 success: this.reflect_properties
 69             });
 70         }
 71     },
 72     
 73     reflect_properties: function() {
 74         this.workspace.trigger('properties:loaded');
 75     },
 76     
 77     run: function(force, mdx) {
 78         // Check for automatic execution
 79 
 80         
 81         if ( (this.properties.properties['saiku.olap.query.automatic_execution'] == "false") &&
 82             ! (force === true)) {
 83             return;
 84         }
 85         Saiku.ui.unblock();
 86 
 87         // TODO - Validate query
 88         // maybe we should sync it with the backend query JSON?
 89         // this definitely needs improvement
 90         var rows = 0;
 91         var columns = 0;
 92         if (this.get('type') != "MDX") {
 93             if (Settings.MODE == "view" || Settings.MODE == "table") {
 94                 var axes = this.get('axes');
 95                 if (axes) {
 96                     for (var axis_iter = 0; axis_iter < axes.length; axis_iter++) {
 97                         var axis = axes[axis_iter];
 98                         if (axis.name && axis.name == "ROWS") {
 99                             rows = axis.dimensionSelections.length;
100                         }
101                         if (axis.name && axis.name == "COLUMNS") {
102                             columns = axis.dimensionSelections.length;
103                         }
104                     }
105                 }
106             } else {
107                 rows = $(this.workspace.el).find('.rows ul li').size();
108                 columns = $(this.workspace.el).find('.columns ul li').size(); 
109             }
110 
111             if (rows == 0 || columns == 0) {
112                 $(this.workspace.el).find('.workspace_results table')
113                     .html('<tr><td><span class="i18n">You need to put at least one level or measure on Columns and Rows for a valid query.</td></tr>');
114                 return;
115             }
116         }
117 
118         // Run it
119         $(this.workspace.table.el)
120             .html('<tr><td>Running query...</td></tr>');
121             // <a class="cancel" href="#cancel">x</a>
122         if (this.get('type')  == "MDX" && mdx != null) {
123             this.result.save({ mdx: mdx });
124         } else {
125             this.result.fetch();
126         }
127     },
128     
129     move_dimension: function(dimension, target, index) {
130         $(this.workspace.el).find('.run').removeClass('disabled_toolbar');
131         var url = "/axis/" + target + "/dimension/" + dimension;
132         
133         this.action.post(url, {
134             data: {
135                 position: index
136             },
137             
138             success: function() {
139                 if (this.query.properties
140                     .properties['saiku.olap.query.automatic_execution'] === 'true') {
141                     this.query.run();
142                 }
143             }
144         });
145     },
146     
147     url: function() {
148         return encodeURI(Saiku.session.username + "/query/" + this.uuid);
149     }
150 });
