 21 /**
 22  * The analysis workspace
 23  */
 24 var Workspace = Backbone.View.extend({
 25     className: 'tab_container',
 26     
 27     events: {
 28         'click .sidebar_separator': 'toggle_sidebar',
 29         'change .cubes': 'new_query',
 30         'drop': 'remove_dimension',
 31         'click .refresh_cubes' : 'refresh'
 32     },
 33     
 34     initialize: function(args) {
 35         // Maintain `this` in jQuery event handlers
 36         _.bindAll(this, "adjust", "toggle_sidebar", "prepare", "new_query", 
 37                 "init_query", "update_caption", "populate_selections","refresh");
 38                 
 39         // Attach an event bus to the workspace
 40         _.extend(this, Backbone.Events);
 41         this.loaded = false;
 42         this.bind('dimensions:loaded',this.populate_selections);
 43 
 44         // Generate toolbar and append to workspace
 45         this.toolbar = new WorkspaceToolbar({ workspace: this });
 46         this.toolbar.render();
 47         
 48         // Create drop zones
 49         this.drop_zones = new WorkspaceDropZone({ workspace: this });
 50         this.drop_zones.render();
 51         
 52         // Generate table
 53         this.table = new Table({ workspace: this });
 54         
 55         // Pull query from args
 56         if (args && args.query) {
 57             this.query = args.query;
 58             this.query.workspace = this;
 59             this.query.save({}, { success: this.init_query });            
 60         }
 61         // Flash cube navigation when rendered
 62         Saiku.session.bind('tab:add', this.prepare);
 63     },
 64     
 65     caption: function() {
 66         if (this.query && this.query.name) {
 67             return this.query.name;
 68         }
 69         
 70         return "Unsaved query (" + (Saiku.tabs.queryCount + 1) + ")";
 71     },
 72     
 73     template: function() {
 74         var template = $("#template-workspace").html() || "";
 75         return _.template(template)({
 76             cube_navigation: Saiku.session.sessionworkspace.cube_navigation
 77         });        
 78     },
 79 
 80     refresh: function() {
 81         Saiku.session.sessionworkspace.refresh();
 82     },
 83     
 84     render: function() {
 85         // Load template
 86         $(this.el).html(this.template());
 87         
 88         // Show toolbar
 89         $(this.el).find('.workspace_toolbar').append($(this.toolbar.el));
 90         
 91         // Show drop zones
 92         $(this.el).find('.workspace_editor').append($(this.drop_zones.el));
 93 
 94         
 95         // Activate sidebar for removing elements
 96         $(this.el).find('.sidebar')
 97             .droppable({
 98                 accept: '.d_measure, .d_dimension'
 99             });
100         
101         // Add results table
102         $(this.el).find('.workspace_results')
103             .append($(this.table.el));
104         
105         // Adjust tab when selected
106         this.tab.bind('tab:select', this.adjust);
107         $(window).resize(this.adjust);
108             
109         // Fire off new workspace event
110         Saiku.session.trigger('workspace:new', { workspace: this });
111 
112         return this; 
113     },
114     
115     clear: function() {
116         // Prepare the workspace for a new query
117         $(this.el).find('.workspace_results table,.connectable')
118             .html('');
119             
120         // Trigger clear event
121         Saiku.session.trigger('workspace:clear', { workspace: this });
122 
123     },
124     
125     adjust: function() {
126         // Adjust the height of the separator
127         $separator = $(this.el).find('.sidebar_separator');
128         var heightReduction = 87;
129         if (Settings.PLUGIN == true || Settings.BIPLUGIN == true) {
130             heightReduction = 2;
131         }
132         $separator.height($("body").height() - heightReduction);
133         $(this.el).find('.sidebar').height($("body").height() - heightReduction);
134         
135         // Adjust the dimensions of the results window
136         $(this.el).find('.workspace_results').css({
137             height: $(document).height() - $("#header").height() -
138                 $(this.el).find('.workspace_toolbar').height() - 
139                 $(this.el).find('.workspace_fields').height() - 40
140         });
141         
142         // Fire off the adjust event
143         this.trigger('workspace:adjust', { workspace: this });
144     },
145     
146     toggle_sidebar: function() {
147         // Toggle sidebar
148         $(this.el).find('.sidebar').toggleClass('hide');
149         $(this.toolbar.el).find('.toggle_sidebar').toggleClass('on');
150         var new_margin = $(this.el).find('.sidebar').hasClass('hide') ?
151                 5 : 265;
152         $(this.el).find('.workspace_inner').css({ 'margin-left': new_margin });
153     },
154     
155     prepare: function() {
156         // Draw user's attention to cube navigation
157         $(this.el).find('.cubes')
158             .parent()
159             .css({ backgroundColor: '#AC1614' })
160             .delay(500)
161             .animate({ backgroundColor: '#fff' }, 'slow');
162     },
163     
164     new_query: function() {
165         // Delete the existing query
166         if (this.query) {
167             this.query.destroy();
168         }
169         
170         // Initialize the new query
171         this.selected_cube = $(this.el).find('.cubes').val();
172         var parsed_cube = this.selected_cube.split('/');
173         var cube = parsed_cube[3];
174         for (var i = 4; i < parsed_cube.length; i++) {
175             cube += "/" + parsed_cube[i];
176         }
177         this.query = new Query({
178             connection: parsed_cube[0],
179             catalog: parsed_cube[1],
180             schema: (parsed_cube[2] == "null" ? "" : parsed_cube[2]) ,
181             cube: decodeURIComponent(cube)
182         }, {
183             workspace: this
184         });
185         
186         // Save the query to the server and init the UI
187         Saiku.session.trigger('workspace:clear', { workspace: this });
188         this.query.save();
189         this.init_query();
190     },
191     
192     init_query: function() {
193         if ((Settings.MODE == "table") && this.query) {
194             this.query.run();
195             return;
196         }
197 
198         if (this.query.get('type') == "MDX") {
199             $(this.drop_zones.el).remove();
200             this.toolbar.switch_to_mdx();
201             if (! $(this.el).find('.sidebar').hasClass('hide')) {
202                 this.toggle_sidebar();
203             }
204 
205             $(this.el).find('.workspace_results').css({
206             height: $(document).height() - (typeof $("#header").height() !== "undefined" ? $("#header").height() : 0)  -
207                 (typeof $(this.el).find('.workspace_toolbar').height() !== "undefined" ? $(this.el).find('.workspace_toolbar').height() : 0)  -
208                 $(this.el).find('.mdx_input').height() - 80
209             });
210 
211         } else {
212             $(this.el).find('.workspace_fields').show();
213             $(this.el).find('.workspace_editor .mdx_input').val('').addClass('hide');
214             $(this.toolbar.el).find('.auto, ,.toggle_fields, .query_scenario, .buckets, .non_empty, .swap_axis, .mdx, .switch_to_mdx').parent().show();
215             $(this.el).find('.run').attr('href','#run_query');
216         }
217 
218         if ((Settings.MODE == "view") && this.query) {
219             this.query.run();
220             return;
221         }
222 
223         // Find the selected cube
224         if (this.selected_cube === undefined) {
225             var schema = this.query.get('schema');
226             this.selected_cube = this.query.get('connection') + "/" + 
227                 this.query.get('catalog') + "/"
228                 + ((schema == "" || schema == null) ? "null" : schema) 
229                 + "/" + this.query.get('cube');
230             $(this.el).find('.cubes')
231                 .val(this.selected_cube);
232         }        
233         // Clear workspace
234         this.clear();
235         
236         if (this.selected_cube) {
237             // Create new DimensionList and MeasureList
238             this.dimension_list = new DimensionList({
239                 workspace: this,
240                 dimension: Saiku.session.sessionworkspace.dimensions[this.selected_cube]
241             });        
242             $(this.el).find('.dimension_tree').html('').append($(this.dimension_list.el));
243             
244             this.measure_list = new DimensionList({
245                 workspace: this,
246                 dimension: Saiku.session.sessionworkspace.measures[this.selected_cube]
247             });
248             $(this.el).find('.measure_tree').html('').append($(this.measure_list.el));
249         } else {
250             // Someone literally selected "Select a cube"
251             $(this.el).find('.dimension_tree').html('');
252             $(this.el).find('.measure_tree').html('');
253             return;
254         }
255 
256     },
257 
258     populate_selections: function(dimension_el) {
259 
260         if (this.other_dimension) {
261         // Populate selections - trust me, this is prettier than it was :-/
262         var axes = this.query ? this.query.get('axes') : false;
263         if (axes) {
264             for (var axis_iter = 0; axis_iter < axes.length; axis_iter++) {
265                 var axis = axes[axis_iter];
266                 var $axis = $(this.el).find('.' + 
267                     axis.name.toLowerCase() + ' ul');
268                 for (var dim_iter = 0; dim_iter < axis.dimensionSelections.length; dim_iter++) {
269                     var dimension = axis.dimensionSelections[dim_iter];
270                     var levels = [];
271                     var members = {};
272                     for (var sel_iter = 0; sel_iter < dimension.selections.length; sel_iter++) {
273                         var selection = dimension.selections[sel_iter];
274                         
275                         // Drag over dimensions and measures
276                         var type, name;
277                         if (selection.dimensionUniqueName == "Measures") {
278                             type = "measure";
279                             name = selection.uniqueName;
280                         } else {
281                             type = "dimension";
282                             name = selection.levelUniqueName;
283                         }
284                             
285                         if (levels.indexOf(name) === -1) {
286                             var $dim = $(dimension_el)
287                                 .find('a[rel="' + name + '"]')
288                                 .parent();
289                             
290                             if (!$dim.html() || $dim.html() == null) {
291                                 $dim = $(this.other_dimension)
292                                 .find('a[rel="' + name + '"]')
293                                 .parent();
294                             }
295                             var $clone = $dim.clone()
296                                 .addClass('d_' + type)
297                                 .appendTo($axis);
298                             
299                             if (type == "dimension") {
300                                 $("<span />").addClass('sprite selections')
301                                     .prependTo($clone);
302                                 $icon = $("<span />").addClass('sort');
303                                 var sort = false;
304                                 _.each(axes, function(i_axis) {
305                                     if (i_axis.sortLiteral && i_axis.sortLiteral != null && i_axis.sortLiteral.indexOf(selection.hierarchyUniqueName) != -1) {
306                                         $icon.addClass(i_axis.sortOrder);
307                                         sort = true;
308                                     }
309                                 });
310                                 if (!sort) {
311                                     $icon.addClass('none');
312                                 }
313                                 
314                                 $icon.insertBefore($clone.find('a'));
315                             }
316 
317                             if (type == "measure") {
318                                 $icon = $("<span />").addClass('sort');
319                                 var sort = false;
320                                 _.each(axes, function(i_axis) {
321                                     if (i_axis.sortLiteral && i_axis.sortLiteral != null && i_axis.sortLiteral.indexOf(name) != -1) {
322                                         $icon.addClass(i_axis.sortOrder);
323                                         sort = true;
324                                     }
325                                 });
326                                 if (!sort) {
327                                     $icon.addClass('none');
328                                 }
329                                 
330                                 $icon.insertBefore($clone.find('a'));
331                             }
332 
333                             
334                             
335                             $dim.css({fontWeight: "bold"})
336                                 .draggable('disable')
337                                 .parents('.parent_dimension')
338                                 .find('.folder_collapsed')
339                                 .css({fontWeight: "bold"}); 
340                             levels.push(name);
341                         }
342                         
343                         // FIXME - something needs to be done about selections here
344                     }
345                 }
346             }
347             
348             this.query.run(true);
349         }
350         
351         // Make sure appropriate workspace buttons are enabled
352         this.trigger('query:new', { workspace: this });
353         
354         // Update caption when saved
355         this.query.bind('query:save', this.update_caption);
356         } else {
357             this.other_dimension = dimension_el;
358         }
359 
360     },
361     
362     update_caption: function() {
363         var caption = this.query.get('name');
364         $(this.tab.el).find('a').html(caption);
365     },
366     
367    
368     
369     remove_dimension: function(event, ui) {
370         this.drop_zones.remove_dimension(event, ui);
371     }
372 });
