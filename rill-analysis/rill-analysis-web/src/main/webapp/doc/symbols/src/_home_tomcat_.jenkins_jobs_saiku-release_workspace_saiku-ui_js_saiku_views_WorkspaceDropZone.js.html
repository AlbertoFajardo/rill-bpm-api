 22 /**
 23  * Sets up workspace drop zones for DnD and other interaction
 24  */
 25 var WorkspaceDropZone = Backbone.View.extend({
 26     template: function() {
 27         var template = $("#template-workspace-dropzones").html() || "";
 28         return _.template(template)();
 29     },
 30     
 31     events: {
 32         'sortbeforestop': 'select_dimension',
 33         'click .d_dimension span.selections': 'selections',
 34         'click .d_measure a' : 'remove_dimension',
 35         'click .d_measure span.sort' : 'sort_measure',
 36         'click .d_dimension span.sort' : 'sort_measure'
 37     },
 38     
 39     initialize: function(args) {
 40         // Keep track of parent workspace
 41         this.workspace = args.workspace;
 42         
 43         // Maintain `this` in jQuery event handlers
 44         _.bindAll(this, "select_dimension", "move_dimension", 
 45                 "remove_dimension", "update_selections","sort_measure");
 46     },
 47     
 48     render: function() {
 49         // Generate drop zones from template
 50         $(this.el).html(this.template());
 51         
 52         // Activate drop zones
 53         $(this.el).find('.connectable').sortable({
 54             connectWith: $(this.el).find('.connectable'),
 55             cursorAt: {
 56                 top: 10,
 57                 left: 35
 58             },
 59             forcePlaceholderSize: true,
 60             items: '> li',
 61             opacity: 0.60,
 62             placeholder: 'placeholder',
 63             tolerance: 'pointer',
 64             
 65             start: function(event, ui) {
 66                 ui.placeholder.text(ui.helper.text());
 67             }
 68 
 69         });
 70         
 71         return this; 
 72     },
 73 
 74     sort_measure: function(event, ui) {
 75         $axis = $(event.target.parentElement).parents('.fields_list_body');
 76         var source = "";
 77         var target = "ROWS";
 78         
 79         if ($axis.hasClass('rows')) { source = "ROWS"; target= "COLUMNS"; }
 80         if ($axis.hasClass('columns')) { source = "COLUMNS"; target="ROWS"; }
 81         if ($axis.hasClass('filter')) { source = "FILTER"; target="COLUMNS"; }
 82 
 83         var sortOrder = "";
 84         if ($(event.target).hasClass('none')) sortOrder = "none";
 85         if ($(event.target).hasClass('BASC')) sortOrder = "BASC";
 86         if ($(event.target).hasClass('BDESC')) sortOrder = "BDESC";
 87 
 88         var memberpath = $(event.target.parentElement).find('a').attr('href').replace('#', '').split('/');
 89         var member = "-";
 90         if ($(event.target).parent().hasClass('d_dimension')) {
 91             member = memberpath[2] + ".CurrentMember.Name";
 92             $axis.find('.d_dimension .sort').removeClass('BASC').removeClass('BDESC').addClass('none');
 93             $axis.parent().parent().find("." + target.toLowerCase() + " .d_measure .sort").removeClass('BASC').removeClass('BDESC').addClass('none');
 94             target = source;
 95         } else {
 96             member = memberpath[memberpath.length -1];
 97             $axis.find('.d_measure .sort').removeClass('BASC').removeClass('BDESC').addClass('none');
 98             $axis.parent().parent().find("." + target.toLowerCase() + " .d_dimension .sort").removeClass('BASC').removeClass('BDESC').addClass('none');
 99         }
100 
101         var futureSortOrder = "none";
102         if (sortOrder == "none") futureSortOrder = "BASC";
103         if (sortOrder == "BASC") futureSortOrder = "BDESC";
104         if (sortOrder == "BDESC") futureSortOrder = "none";
105         
106 
107 
108         $(event.target).removeClass('none').addClass(futureSortOrder);
109 
110         if (futureSortOrder == "none") {
111             var url = "/axis/" + target + "/sort/";
112             this.workspace.query.action.del(url, {
113                 success: this.workspace.query.run
114             });
115         } else {
116             var url = "/axis/" + target + "/sort/" + futureSortOrder + "/" + member;
117             this.workspace.query.action.post(url, {
118                 success: this.workspace.query.run
119             });
120         }
121 
122     },
123     
124     select_dimension: function(event, ui) {
125 
126         $axis = ui.item.parents('.fields_list_body');
127         var target = "";
128         
129         if ($axis.hasClass('rows')) target = "ROWS";
130         if ($axis.hasClass('columns')) target = "COLUMNS";
131         if ($axis.hasClass('filter')) target = "FILTER";
132 
133 
134         // Short circuit if this is a move
135         if (ui.item.hasClass('d_measure') ||
136                 ui.item.hasClass('d_dimension')) {
137             this.move_dimension(event, ui, target);
138             return;
139         }
140 
141         // Make the element and its parent bold
142         var original_href = ui.item.find('a').attr('href');
143         var $original = $(this.workspace.el).find('.sidebar')
144             .find('a[href="' + original_href + '"]').parent('li');
145         $original
146             .css({fontWeight: "bold"})
147             .draggable('disable');
148         $original.parents('.parent_dimension')
149             .find('.folder_collapsed')
150             .css({fontWeight: "bold"});
151         
152 
153         // Wrap with the appropriate parent element
154         if (ui.item.find('a').hasClass('dimension')) {
155             var $icon = $("<div />").addClass('sprite').addClass('selections');
156             var $icon2 = $("<span />").addClass('sprite').addClass('sort none');
157         
158             ui.item.addClass('d_dimension').prepend($icon);
159             ui.item.addClass('d_dimension').prepend($icon2);
160         } else {
161             var $icon = $("<span />").addClass('sort none');
162             ui.item.addClass('d_measure').prepend($icon);
163         }
164 
165 
166 
167         var member = ui.item.find('a').attr('href').replace('#', '');
168         var dimension = member.split('/')[0];
169         var dimensions = [];
170 
171         this.update_selections(event,ui);
172 
173         $axis.find('a').each( function(i,element) {
174             var imember = $(element).attr('href');
175             var idimension = imember.replace('#', '').split('/')[0]; 
176             if (dimensions.indexOf(idimension) == -1) {
177                 dimensions.push(idimension);
178             }
179         });
180 
181         var index = dimensions.indexOf(dimension);
182 
183 
184         // Notify the model of the change
185         this.workspace.query.move_dimension(member, 
186                 target, index);
187 
188         // Prevent workspace from getting this event
189         return true;
190     },
191     
192     move_dimension: function(event, ui, target) {
193         if (! ui.item.hasClass('deleted')) {
194             $axis = ui.item.parents('.fields_list_body');
195 
196             // Notify the model of the change
197             var dimension = ui.item.find('a').attr('href').replace('#', '').split('/')[0];
198             var dimensions = [];
199 
200             this.update_selections(event,ui);
201 
202             $axis.find('a').each( function(i,element) {
203                 var imember = $(element).attr('href');
204                 var idimension = imember.replace('#', '').split('/')[0]; 
205                 if (dimensions.indexOf(idimension) == -1) {
206                     dimensions.push(idimension);
207                 }
208             });
209             var index = dimensions.indexOf(dimension);
210 
211             
212                 this.workspace.query.move_dimension(dimension, 
213                     target, index);
214         }
215         
216         // Prevent workspace from getting this event
217         event.stopPropagation();
218         return false;
219     },
220 
221     update_selections: function(event, ui) {
222         var member = ui.item.find('a').attr('href');
223         var dimension = member.replace('#', '').split('/')[0];
224         var index = ui.item.parent('.connectable').children().index(ui.item);
225         var axis = ui.item.parents('.fields_list_body');
226         var allAxes = axis.parent().parent();
227         var target = '';
228         var source = '';
229         var myself = this;
230         var $originalItem =  $(myself.workspace.el).find('.sidebar')
231                                     .find('a[href="' + member + '"]').parent();
232         var insertElement = $(ui.item);
233         var type = $(ui.item).hasClass('d_dimension') ? "d_dimension" : "d_measure";
234 
235         var sourceAxis = "";
236         if ($axis.hasClass('rows')) sourceAxis = "ROWS";
237         if ($axis.hasClass('columns')) sourceAxis = "COLUMNS";
238         if ($axis.hasClass('filter')) sourceAxis = "FILTER";
239 
240         source = ".rows, .columns, .filter";
241         allAxes.find(source).find('a').each( function(index, element) {
242             var p_member = $(element).attr('href').replace('#', '');
243             var p_dimension = p_member.split('/')[0];
244             if (p_dimension == dimension && (( "#" + p_member) != member )) {
245                 $(element).parent().remove();
246             }
247         });
248         
249         var n_dimension = null;
250         var p_dimension = null;
251 
252         
253         var prev = $(ui.item).prev();
254         if (prev && prev.length > 0) {
255             var p_member = prev.find('a').attr('href');
256             p_dimension = p_member.replace('#', '').split('/')[0];
257         }
258         var next = $(ui.placeholder).next();
259 
260         while (p_dimension != null && next && next.length > 0 ) {
261             var n_member = next.find('a').attr('href');
262             n_dimension = n_member.replace('#', '').split('/')[0]; 
263             if (p_dimension == n_dimension) {
264                 next.insertBefore($(ui.item));
265             } else {
266                 p_dimension = null;
267             }
268             next = $(ui.placeholder).next();
269         }
270             
271 
272         $originalItem.parent().find('.ui-draggable-disabled').clone().attr('class', 'ui-draggable').removeAttr('style')
273                                 .addClass(type)
274                                 .insertAfter(insertElement);
275         
276 
277         axis.find('.d_dimension a').each( function(index, element) {
278             element = $(element);
279             if (!element.prev() || (element.prev() && element.prev().length == 0)) {
280                 var $icon = $("<span />").addClass('sprite').addClass('selections');
281                 $icon.insertBefore(element);
282                 var $icon = $("<span />").addClass('sprite').addClass('sort none');
283                 $icon.insertBefore(element);
284             }
285         });
286 
287         axis.find('.d_measure a, .d_dimension a').each( function(index, element) {
288             element = $(element);
289             if (!element.prev() || (element.prev() && element.prev().length == 0)) {
290                 if (sourceAxis != "FILTER") {
291                     var $icon = $("<span />").addClass('sort none');
292                     $icon.insertBefore(element);
293                 }
294                 
295             }
296         });
297 
298         $(ui.item).remove();
299 
300     },
301 
302 
303     remove_dimension: function(event, ui) {
304         // Reenable original element
305         var $source = ui ? ui.draggable : $(event.target).parent();
306         var original_href = $source.find('a').attr('href');
307         var $original = $(this.workspace.el).find('.sidebar')
308             .find('a[href="' + original_href + '"]').parent('li');
309         $original
310             .draggable('enable')
311             .css({ fontWeight: 'normal' });
312         
313         // Unhighlight the parent if applicable
314         if ($original.parents('.parent_dimension')
315                 .children().children('.ui-state-disabled').length === 0) {
316             $original.parents('.parent_dimension')
317                 .find('.folder_collapsed')
318                 .css({fontWeight: "normal"});
319         }
320         
321         // Notify server
322         var target = '';
323         var dimension = original_href.replace('#', '');
324         $target_el = $source.parent().parent('div.fields_list_body');
325         if ($target_el.hasClass('rows')) target = "ROWS";
326         if ($target_el.hasClass('columns')) target = "COLUMNS";
327         if ($target_el.hasClass('filter')) target = "FILTER";
328         
329         var url = "/axis/" + target + "/dimension/" + dimension;
330         this.workspace.query.action.del(url, {
331             success: this.workspace.query.run
332         });
333         
334         // Remove element
335         $source.addClass('deleted').remove();
336         
337         // Prevent workspace from getting this event
338         event.stopPropagation();
339         event.preventDefault();
340         return false;
341     },
342     
343     selections: function(event, ui) {
344         // Determine dimension
345         var $target = $(event.target).hasClass('sprite') ?
346             $(event.target).parent().find('.dimension') :
347             $(event.target);
348         var key = $target.attr('href').replace('#', '');
349         
350         // Launch selections dialog
351         (new SelectionsModal({
352             target: $target,
353             name: $target.text(),
354             key: key,
355             workspace: this.workspace
356         })).open();
357         
358         // Prevent default action
359         try {
360             event.preventDefault();
361         } catch (e) { }
362         return false;
363     }
364 });
