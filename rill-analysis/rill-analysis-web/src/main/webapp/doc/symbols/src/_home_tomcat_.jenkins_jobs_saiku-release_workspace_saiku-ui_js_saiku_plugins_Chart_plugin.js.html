 22 /**
 23  * Renders a chart for each workspace
 24  */
 25 var Chart = Backbone.View.extend({
 26     initialize: function(args) {
 27         this.workspace = args.workspace;
 28         
 29         // Create a unique ID for use as the CSS selector
 30         this.id = _.uniqueId("chart_");
 31         $(this.el).attr({ id: this.id });
 32         
 33         // Bind table rendering to query result event
 34         _.bindAll(this, "render", "receive_data", "process_data", "show", 
 35             "setOptions");
 36         this.workspace.bind('query:result', this.receive_data);
 37         
 38         // Add chart button
 39         this.add_button();
 40         this.workspace.toolbar.chart = this.show;
 41         
 42         // Listen to adjust event and rerender chart
 43         this.workspace.bind('workspace:adjust', this.render);
 44         
 45         // Create navigation
 46         this.nav = $("<div class='chart-switcher'>" +
 47         		"<a href='#bar' class='i18n'>bar</a>" +
 48                 "<a href='#stackedBar' class='i18n'>stacked bar</a>" +
 49         		"<a href='#line' class='i18n'>line</a>" +
 50         		"<a href='#pie' class='i18n'>pie</a>" +
 51                 "<a href='#heatgrid' class='i18n'>heatgrid</a>" +
 52         		"</div>").css({
 53         		    'padding-bottom': '10px'
 54         		});
 55         this.nav.find('a').css({ 
 56                     color: '#666', 
 57                     'margin-right': '5px', 
 58                     'text-decoration': 'none', 
 59                     'border': '1px solid #ccc', 
 60                     padding: '5px' 
 61                 })
 62                 .click(this.setOptions);
 63     
 64         // Append chart to workspace
 65         $(this.workspace.el).find('.workspace_results')
 66             .prepend($(this.el).hide())
 67             .prepend(this.nav.hide());
 68     },
 69     
 70     add_button: function() {
 71         var $chart_button = 
 72             $('<a href="#chart" class="chart button disabled_toolbar i18n" title="Toggle Chart"></a>')
 73             .css({  'background-image': "url('js/saiku/plugins/Chart/chart.png')",
 74                     'background-repeat':'no-repeat',
 75                     'background-position':'50% 50%'
 76                 });
 77 
 78         var $chart_li = $('<li class="seperator"></li>').append($chart_button);
 79         $(this.workspace.toolbar.el).find("ul").append($chart_li);
 80     },
 81     
 82     show: function(event, ui) {
 83         $(this.workspace.el).find('.workspace_results table').toggle();
 84         $(this.el).toggle();
 85         $(this.nav).toggle();
 86         $(event.target).toggleClass('on');
 87         
 88         if ($(event.target).hasClass('on')) {
 89             this.render();
 90         }
 91     },
 92     
 93     setOptions: function(event) {
 94         var type = $(event.target).attr('href').replace('#', '');
 95         try {
 96             this[type]();
 97         } catch (e) { }
 98         
 99         return false;
100     },
101     
102     stackedBar: function() {
103         this.options.stacked = true;
104         this.options.type = "BarChart";
105         this.render();
106     },
107     
108     bar: function() {
109         this.options.stacked = false;
110         this.options.type = "BarChart";
111         this.render();
112     },
113     
114     line: function() {
115         this.options.stacked = false;
116         this.options.type = "LineChart";
117         this.render();
118     },
119     
120     pie: function() {
121         this.options.stacked = false;
122         this.options.type = "PieChart";
123         this.render();
124     },
125 
126     heatgrid: function() {
127         this.options.stacked = false;
128         this.options.type = "HeatGridChart";
129         this.render();
130     },
131     
132     render: function() {
133         if (! $(this.workspace.toolbar.el).find('.chart').hasClass('on')) {
134             return;
135         }
136         
137         var options = _.extend({
138             canvas: this.id,
139             width: $(this.workspace.el).find('.workspace_results').width() - 20,
140             height: $(this.workspace.el).find('.workspace_results').height() - 40,
141             orientation: 'vertical',
142             stacked: false,
143             animate: false,
144             showValues: false,
145             legend: true,
146             legendPosition:"top",
147             legendAlign: "right",
148             colors: ["#B40010", "#CCC8B4", "#DDB965", "#72839D", "#1D2D40"],
149             type: 'BarChart'
150         }, this.options);
151         
152         if (options.type == "HeatGridChart") {
153             options = _.extend({
154                     canvas: this.id,
155                     width: $(this.workspace.el).find('.workspace_results').width() - 80,
156                     height: $(this.workspace.el).find('.workspace_results').height(),
157                     animate: false,
158                     clickable: false,
159                     orientation: "horizontal",
160                     showValues: false,
161                     showXScale: true,
162                     xAxisPosition: "bottom",
163                     showYScale: true,
164                     panelSizeRatio: 0.8,
165                     yAxisPosition: "left",
166                     yAxisSize: 70,
167                     minColor: "#FEDFE1",
168                     maxColor: "#F11929",
169                     extensionPoints: {
170                         xAxisLabel_textAngle: -(Math.PI / 2),
171                         xAxisLabel_textAlign: "right",
172                         xAxisLabel_bottom: 10
173                     }
174             }, this.options);
175         }
176         if (this.data.resultset.length > 5 ) {
177             options.extensionPoints = {
178                 xAxisLabel_textAngle: -(Math.PI / 2),
179                 xAxisLabel_textAlign: "right",
180                 xAxisLabel_bottom: 10
181             };
182             
183             options.xAxisSize = 100;
184         }
185         
186         this.chart = new pvc[options.type](options);
187         
188         this.chart.setData(this.data, {
189             crosstabMode: true,
190             seriesInRows: false
191         });
192         
193         try {
194             this.chart.render();
195             Saiku.i18n.automatic_i18n();
196         } catch (e) {
197             $(this.el).text("Could not render chart");
198         }
199     },
200     
201     receive_data: function(args) {
202         return _.delay(this.process_data, 0, args);
203     },
204     
205     process_data: function(args) {
206         this.data = {};
207         this.data.resultset = [];
208         this.data.metadata = [];
209         this.data.height = 0;
210         this.data.width = 0;
211         
212         if (args.data.cellset && args.data.cellset.length > 0) {
213             
214             var lowest_level = 0;
215         
216             for (var row = 0; row < args.data.cellset.length; row++) {
217                 if (args.data.cellset[row][0].type == "ROW_HEADER_HEADER") {
218                     this.data.metadata = [];
219                     for (var field = 0; field < args.data.cellset[row].length; field++) {
220                         if (args.data.cellset[row][field].type == "ROW_HEADER_HEADER") {
221                             this.data.metadata.shift();
222                             lowest_level = field;
223                         }
224                         
225                         this.data.metadata.push({
226                             colIndex: field,
227                             colType: typeof(args.data.cellset[row + 1][field].value) !== "number" &&
228                                 isNaN(args.data.cellset[row + 1][field].value
229                                 .replace(/[^a-zA-Z 0-9.]+/g,'')) ? "String" : "Numeric",
230                             colName: args.data.cellset[row][field].value
231                         });
232                     }
233                 } else if (args.data.cellset[row][0].value !== "null" && args.data.cellset[row][0].value !== "") {
234                     var record = [];
235                     this.data.width = args.data.cellset[row].length;
236                     for (var col = lowest_level; col < args.data.cellset[row].length; col++) {
237                         var value = args.data.cellset[row][col].value;
238                         // check if the resultset contains the raw value, if not try to parse the given value
239                         if (args.data.cellset[row][col].properties.raw && args.data.cellset[row][col].properties.raw !== "null")
240                         {
241                             value = parseFloat(args.data.cellset[row][col].properties.raw);
242                         } else if (typeof(args.data.cellset[row][col].value) !== "number" &&
243                             parseFloat(args.data.cellset[row][col].value.replace(/[^a-zA-Z 0-9.]+/g,''))) 
244                         {
245                             value = parseFloat(args.data.cellset[row][col].value.replace(/[^a-zA-Z 0-9.]+/g,''));
246                         }
247                         if (col == lowest_level) {
248                             value += " [" + row + "]";
249                         }
250                         record.push(value);
251                     }
252                     this.data.resultset.push(record);
253                 }
254             }
255             this.data.height = this.data.resultset.length;
256             this.render();
257         } else {
258             $(this.el).text("No results");
259         }
260     }
261 });

263 /**
264  * Loads CCC and initializes chart plugin
265  */
266 (function() {
267     // Initialize CCC
268     $.ajax({
269         url: "js/saiku/plugins/Chart/ccc.js",
270         dataType: "script",
271         cache: true,
272         success: function() {
273 
274             var initPlugin = function(session) {
275                 function new_workspace(args) {
276                     // Add chart element
277                     if (typeof args.workspace.chart == "undefined") {
278                         args.workspace.chart = new Chart({ workspace: args.workspace });
279                     } 
280                 }
281 
282                 function clear_workspace(args) {
283                     if (typeof args.workspace.chart != "undefined") {
284                         $(args.workspace.chart.nav).hide();
285                         $(args.workspace.chart.el).hide();
286                         $(args.workspace.chart.el).parents().find('.workspace_results table').show();
287                     }
288                 }
289                 
290                 // Attach chart to existing tabs
291                 for(var i = 0; i < Saiku.tabs._tabs.length; i++) {
292                     var tab = Saiku.tabs._tabs[i];
293                     new_workspace({
294                         workspace: tab.content
295                     });
296                 };
297                 
298                 // Attach chart to future tabs
299                 session.bind("workspace:new", new_workspace);
300                 session.bind("workspace:clear", clear_workspace);
301             };
302 
303             if (typeof Saiku.session == "undefined") {
304                 Saiku.events.bind('session:new', initPlugin);
305             } else {
306                 initPlugin(Saiku.session);
307             }
308         }
309     });
310 }());
