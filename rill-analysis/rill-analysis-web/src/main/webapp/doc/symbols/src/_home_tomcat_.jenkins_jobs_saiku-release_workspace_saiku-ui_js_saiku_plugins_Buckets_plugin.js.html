 22 /**
 23  * Allow the creation of buckets for queries
 24  */
 25 var Buckets = Backbone.View.extend({
 26 
 27     events: {
 28         'click .add_bucket': 'add_bucket',
 29         'click  a.save' : 'save_bucket',
 30         'click .bucket' : 'click_bucket',
 31         'click .delete' : 'delete_bucket'
 32     },
 33 
 34     bucket_css: { 
 35         'color': '#555', 
 36         'margin-right': '5px', 
 37         'text-decoration': 'none', 
 38         'border': '1px solid #ccc', 
 39         'padding': '5px',
 40         '-moz-border-radius': '3px',
 41         '-webkit-border-radius': '3px'
 42     },
 43 
 44     tags: [],
 45 
 46 
 47      tags_template: function() {
 48             var self = this;
 49             var t = "<div class='bucket_items'><ul>" +
 50                 "<li><a href='#' class='add_bucket i18n button' title='Add new tag by selecting cells from your result!'> </a></li>";
 51             _.each(this.tags, function(tag) {
 52               t += self.tag_template(tag);
 53             });     
 54             t += "</div>";
 55             return t;
 56     },
 57 
 58     tag_template: function(tag) {
 59           var title = tag.name + ": ";
 60         _.each(tag.saikuTuples, function(tuple) {
 61             var first = true;
 62             title += " (";
 63             _.each(tuple.saikuMembers, function(member) {
 64                     if (!first) {
 65                         title += ",  ";
 66                     }
 67                     first = false;
 68                     title += member.uniqueName;
 69             });
 70             title += ")";
 71         });
 72 
 73           return "<li class='seperator'><a href='#" + tag.name + "' title='" + title + "' class='bucket button'>" + tag.name + "</a></li>" 
 74           + "<li style='padding-left:0px'><a class='delete' href='#" + tag.name + "'>x</a></li>";
 75     },
 76     
 77 
 78     initialize: function(args) {
 79         this.workspace = args.workspace;
 80         
 81         // Create a unique ID for use as the CSS selector
 82         this.id = _.uniqueId("buckets_");
 83         $(this.el).attr({ id: this.id });
 84         
 85         // Bind table rendering to query result event
 86         _.bindAll(this, "render",  "show", "buildTemplate", "render", "deactivate_add_bucket", 
 87             "add_bucket", "save_bucket", "click_bucket");
 88         
 89         // Add buckets button
 90         this.add_button();
 91         this.workspace.toolbar.buckets = this.show;
 92         
 93         // Listen to adjust event and rerender buckets
 94         this.workspace.bind('workspace:adjust', this.render);
 95         
 96         $(this.workspace.el).find('.workspace_results')
 97             .prepend($(this.el).hide());
 98 
 99 
100     },
101     
102     add_button: function() {
103         var $stats_button = 
104             $('<a href="#buckets" class="buckets button disabled_toolbar i18n" title="Tags"></a>')
105             .css({  'background-image': "url('js/saiku/plugins/Buckets/tag_red.png')",
106                     'background-repeat':'no-repeat',
107                     'background-position':'50% 50%'
108                 });
109 
110         var $stats_li = $('<li class="seperator"></li>').append($stats_button);
111         $(this.workspace.toolbar.el).find("ul").append($stats_li);
112     },
113     
114     show: function(event, ui) {
115         var self = this;
116         $(this.el).toggle();
117         $(event.target).toggleClass('on');
118         
119         if ($(event.target).hasClass('on')) {
120             var schema = self.workspace.query.get('schema');
121             var cube = self.workspace.query.get('connection') + "-" + 
122                     self.workspace.query.get('catalog') + "-"
123                     + ((schema == "" || schema == null) ? "null" : schema) 
124                     + "-" + self.workspace.query.get('cube');
125 
126 
127             this.repo = new TagRepository({
128                 cube: cube
129             }).fetch({success: this.buildTemplate});
130         } else {
131             this.workspace.query.action.del("/tag", { 
132                             success: this.workspace.query.run
133             });
134         }
135         
136     },
137 
138     buildTemplate: function(model,response) {
139         this.tags = response;
140         this.render();
141     },
142     
143     render: function() {
144         if (! $(this.workspace.toolbar.el).find('.buckets').hasClass('on')) {
145             return;
146         }
147 
148         $(this.el).empty();
149 
150         
151         var rendered = this.tags_template();
152         var $table = $(rendered);
153         $(this.el).append($table)
154         
155     },
156 
157     deactivate_add_bucket: function() {
158         var self = this;
159         var $addBtn = $(self.el).find('.add_bucket');
160         $addBtn.removeClass('on');
161         $(self.el).find('.new_bucket').parent().remove();
162         return;
163     },
164 
165     add_bucket: function(event) {
166         var self = this;
167         var $addBtn = $(self.el).find('.add_bucket');
168         if ($addBtn.hasClass('on')) {
169             self.deactivate_add_bucket();
170             return;
171         }
172         $addBtn.addClass('on');
173 
174         
175         $("<li><input id='new_bucket' type='text' class='new_bucket'/></li><li>" 
176             + "<a href='#save_bucket' class='i18n save sprite button new_bucket' title='Save Tag'></a></li>")
177                             .insertAfter($(self.el).find('.bucket_items .add_bucket').parent());
178 
179         var clicked = function(event) {
180             $target = $(event.target).hasClass('data') ?
181                             $(event.target).find('div') : $(event.target);
182 
183             if ($target.parent().hasClass('selected')) {
184                 $target.parent().removeClass('selected');
185             } else {
186                 $target.parent().addClass('selected');
187             }
188         };
189 
190         $(self.workspace.el).find("td.data").addClass('cellhighlight').unbind('click').click(clicked);
191         $(self.workspace.el).find(".query_scenario, .drillthrough, .drillthrough_export").removeClass('on');
192     },
193 
194     save_bucket: function() {
195         var self = this;
196         var $cells = $(self.workspace.el).find("td.selected div");
197         if($cells.size() < 1) {
198             alert("You need to select at least 1 cell for tagging before you can save!");
199             return;
200         }
201         var positions = "";
202         $.each($cells, function(index,element) {
203             if (index > 0) {
204                 positions += ",";
205             }
206             positions += $(element).attr('rel');
207             
208         });
209 
210         $(self.workspace.el).find("td.data").removeClass('cellhighlight').unbind('click');
211         $(self.workspace.el).find("td.selected").removeClass('selected');
212 
213         var tagname = $(self.el).find('#new_bucket').val();
214         
215         var saveBucket = function(model, response) {
216             self.tags.push(model);
217             self.deactivate_add_bucket();
218                 $tag = $(self.tag_template(model))
219                             .appendTo($(self.el).find('.bucket_items ul'));
220         };
221 
222         var schema = self.workspace.query.get('schema');
223         var cube = self.workspace.query.get('connection') + "-" + 
224                 self.workspace.query.get('catalog') + "-"
225                 + ((schema == "" || schema == null) ? "null" : schema) 
226                 + "-" + self.workspace.query.get('cube');
227 
228         (new SaikuTag({
229             positions: positions,
230             name: tagname,
231             cube: cube,
232             queryname: self.workspace.query.id
233         },{success: saveBucket})).save();
234 
235     },
236     click_bucket: function(event) {
237         var tagName = $(event.target).attr('href').replace('#','');
238         var self = this;
239         if ($(event.target).hasClass('on')) {
240             $(event.target).removeClass('on');
241             this.workspace.query.action.del("/tag", { 
242                             success: this.workspace.query.run
243             });
244         } else {
245             $(event.target).addClass('on');
246 
247             _.each(this.tags, function(tag) {
248                 if (tag.name == tagName) {
249                     self.workspace.query.action.put("/tag", { 
250                           success: self.workspace.query.run, data: {tag:JSON.stringify(tag)}});
251                 }
252             });     
253         }
254         $(event.target).parent().siblings().find('.on').removeClass('on');
255     },
256 
257     delete_bucket: function(event) {
258         var tagname = $(event.target).attr('href').replace('#','');
259         var self = this;
260         var al= function() {
261             alert('y');
262         };
263 
264         var schema = self.workspace.query.get('schema');
265         var cube = self.workspace.query.get('connection') + "-" + 
266                 self.workspace.query.get('catalog') + "-"
267                 + ((schema == "" || schema == null) ? "null" : schema) 
268                 + "-" + self.workspace.query.get('cube');
269 
270 
271         (new SaikuTag({
272             name: tagname,
273             id: 'dummy',
274             cube: cube
275         },{})).destroy();
276 
277         $(event.target).parent().prev().remove();
278         $(event.target).parent().remove();
279     }
280 });

283  * Start Plugin
284  */ 
285  Saiku.events.bind('session:new', function(session) {
286 
287         function new_workspace(args) {
288             // Add stats element
289             if (typeof args.workspace.buckets == "undefined") {
290                 args.workspace.buckets = new Buckets({ workspace: args.workspace });
291                 args.workspace.bind('query:result', args.workspace.buckets.deactivate_add_bucket);
292             }
293         }
294 
295         function clear_workspace(args) {
296                     if (typeof args.workspace.buckets != "undefined") {
297                         $(args.workspace.buckets.el).hide();
298 
299                     }
300         }
301         
302         // Attach stats to existing tabs
303         for(var i = 0; i < Saiku.tabs._tabs.length; i++) {
304             var tab = Saiku.tabs._tabs[i];
305             new_workspace({
306                 workspace: tab.content
307             });
308         };
309 
310         // Attach stats to future tabs
311         Saiku.session.bind("workspace:new", new_workspace);
312         Saiku.session.bind("workspace:clear", clear_workspace);
313     });

316 var SaikuTag = Backbone.Model.extend({
317     initialize: function(args, options) {
318         _.extend(this.attributes, args);
319         if (options != null && options.success) {
320             this.parse = options.success;
321         }
322 
323     },
324 
325     url: function() {
326         return encodeURI(Saiku.session.username + "/tags/" + this.get('cube') + "/" + this.get('name'));
327     }
328 });

330 /**

332  */
333 var TagRepository = Backbone.Collection.extend({
334     model: SaikuTag,
335     
336     initialize: function(args, options) {
337         this.cube = args.cube;
338     },
339     
340

343     }
