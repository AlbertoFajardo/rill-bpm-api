1 /**
  2  * Dialog for member selections
  3  */
  4 var DrillthroughModal = Modal.extend({
  5     type: "drillthrough",
  6     
  7     buttons: [
  8         { text: "Ok", method: "ok" },
  9         { text: "Cancel", method: "close" }
 10     ],
 11 
 12     events: {
 13         'click .collapsed': 'select',
 14         'click .expand': 'select',
 15         'click .folder_collapsed': 'select',
 16         'click .folder_expanded': 'select',
 17         
 18         'click .dialog_footer a:' : 'call',
 19         'click .parent_dimension input' : 'select_dimension',
 20         'click .measure_tree input' : 'select_measure'
 21     },
 22 
 23     
 24     initialize: function(args) {
 25         // Initialize properties
 26         _.extend(this, args);
 27         this.options.title = args.title;
 28         this.query = args.workspace.query;
 29         
 30         this.position = args.position;
 31         this.action = args.action;
 32         Saiku.ui.unblock();
 33         _.bindAll(this, "ok", "drilled");
 34 
 35         // Resize when rendered
 36         this.bind('open', this.post_render);
 37         this.render();
 38                // Load template
 39        $(this.el).find('.dialog_body')
 40           .html(_.template($("#template-drillthrough").html())(this));
 41         // Show dialog
 42         $(this.el).find('.maxrows').val(this.maxrows);
 43                     
 44         var schema = this.query.get('schema');
 45         var key = this.query.get('connection') + "/" + 
 46                 this.query.get('catalog') + "/"
 47                 + ((schema == "" || schema == null) ? "null" : schema) 
 48                 + "/" + encodeURIComponent(this.query.get('cube'));
 49 
 50         var dimensions = Saiku.session.sessionworkspace.dimensions[key].get('data');
 51         var measures = Saiku.session.sessionworkspace.measures[key].get('data');
 52 
 53         var container = $("#template-drillthrough-list").html();
 54         var templ_dim =_.template($("#template-drillthrough-dimensions").html())({dimensions: dimensions});
 55         var templ_measure =_.template($("#template-drillthrough-measures").html())({measures: measures});
 56 
 57         $(container).appendTo($(this.el).find('.dialog_body'));
 58         $(this.el).find('.sidebar').height(($("body").height() / 2) + ($("body").height() / 6) );
 59         $(this.el).find('.sidebar').width(380);
 60 
 61         $(this.el).find('.dimension_tree').html('').append($(templ_dim));
 62         $(this.el).find('.measure_tree').html('').append($(templ_measure));
 63         
 64     },
 65     
 66     select: function(event) {
 67         var $target = $(event.target).hasClass('root')
 68             ? $(event.target) : $(event.target).parent().find('span');
 69         if ($target.hasClass('root')) {
 70             $target.find('a').toggleClass('folder_collapsed').toggleClass('folder_expand');
 71             $target.toggleClass('collapsed').toggleClass('expand');
 72             $target.parents('li').find('ul').children('li').toggle();
 73         }
 74         
 75         return false;
 76     },
 77 
 78     select_dimension: function(event) {
 79         var $target = $(event.target);
 80         var checked = $target.is(':checked');
 81         $target.parent().find('input').attr('checked', checked);
 82     },
 83 
 84     select_measure: function(event) {
 85         var $target = $(event.target);
 86         var checked = $target.is(':checked');
 87         if(checked) {
 88             $target.parent().siblings().find('input').attr('checked', false);
 89         }
 90     },
 91 
 92 
 93     post_render: function(args) {
 94         $(args.modal.el).parents('.ui-dialog').css({ width: "150px" });
 95     },
 96     
 97     ok: function() {
 98         // Notify user that updates are in progress
 99         var $loading = $("<div>Drilling through...</div>");
100         $(this.el).find('.dialog_body').children().hide();
101         $(this.el).find('.dialog_body').prepend($loading);
102         var selections = ""
103         $(this.el).find('.check_level:checked').each( function(index) { 
104             if (index > 0) {
105                 selections += ", ";
106             }
107             selections += $(this).val();
108         });
109 
110         var maxrows = $(this.el).find('.maxrows').val();
111         var params = "?maxrows=" + maxrows;
112         params = params + (typeof this.position !== "undefined" ? "&position=" + this.position : "" );
113         params += "&returns=" + selections;
114         if (this.action == "export") {
115             var location = Settings.REST_URL +
116                 Saiku.session.username + "/query/" + 
117                 this.query.id + "/drillthrough/export/csv" + params;
118             this.close();
119             window.open(location);
120         } else if (this.action == "table") {
121             Saiku.ui.block("Executing drillthrough...");
122             this.query.action.get("/drillthrough", { data: { position: this.position, maxrows: maxrows , returns: selections}, success: this.drilled } );
123             this.close();
124         }
125         
126         return false;
127     },
128 
129     drilled: function(model, response) {
130         var table = new Table({ workspace: this.workspace });
131         table.render({ data: response }, true);
132 
133         Saiku.ui.unblock();
134         var html = '<div id="fancy_results" class="workspace_results" style="overflow:visible"><table>' + $(table.el).html() + '</table></div>';
135         this.remove();
136         table.remove();
137         $.fancybox(html
138             ,
139             {
140             'autoDimensions'    : false,
141             'autoScale'         : false,
142             'height'            :  ($("body").height() - 100),
143             'width'             :  ($("body").width() - 100),
144             'transitionIn'      : 'none',
145             'transitionOut'     : 'none'
146             }
147         );
148 
149     },
150     
151     finished: function() {
152         $(this.el).dialog('destroy').remove();
153         this.query.run();
154     }
155 });
