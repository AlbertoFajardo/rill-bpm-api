 45 /**
 46  * Extend settings with query parameters
 47  */
 48 Settings.GET = function () {
 49     var qs = document.location.search;
 50     qs = qs.split("+").join(" ");
 51     var params = {},
 52         tokens,
 53         re = /[?&]?([^=]+)=([^&]*)/g;
 54 
 55     while (tokens = re.exec(qs)) {
 56         var value = decodeURIComponent(tokens[2]);
 57         if (! isNaN(value)) value = parseInt(value);
 58         if (value === "true") value = true;
 59         if (value === "false") value = false;
 60         params[decodeURIComponent(tokens[1]).toUpperCase()]
 61             = value;
 62     }
 63 
 64     return params;
 65 }();
 66 _.extend(Settings, Settings.GET);

 73 /**
 74  * < IE9 doesn't support Array.indexOf
 75  */
 76 if (!Array.prototype.indexOf)
 77 {
 78   Array.prototype.indexOf = function(elt /*, from*/)
 79   {
 80     var len = this.length >>> 0;
 81 
 82     var from = Number(arguments[1]) || 0;
 83     from = (from < 0)
 84          ? Math.ceil(from)
 85          : Math.floor(from);
 86     if (from < 0)
 87       from += len;
 88 
 89     for (; from < len; from++)
 90     {
 91       if (from in this &&
 92           this[from] === elt)
 93         return from;
 94     }
 95     return -1;
 96   };
 97 }
